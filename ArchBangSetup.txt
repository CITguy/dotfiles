#1: Getting Dropbox/Chrome to open directories/files correctly
  Solution:
    export DE=xfce in ~/.xinitrc

#2: No route/ifconfig

#3: Permanently map Caps Lock to Escape
  Solution:
    edit /etc/rc.local and add the following:
      echo keycode 58 = Escape | sudo loadkeys -
